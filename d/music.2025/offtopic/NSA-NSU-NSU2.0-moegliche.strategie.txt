class Individual:
    def __init__(self, id, radicalization_level, openness, systemic_support):
        self.id = id
        self.radicalization_level = radicalization_level  # 0..1
        self.openness = openness  # 0..1 (psychologische Bereitschaft)
        self.systemic_support = systemic_support  # 0..1
        self.rehabilitated = False

    def interact(self, intervention_effort):
        if self.rehabilitated:
            return
        reduction = intervention_effort * self.openness * (1 - self.systemic_support)
        self.radicalization_level -= reduction
        if self.radicalization_level < 0.2:
            self.rehabilitated = True

class Group:

        self.members = members

    def recruit(self):

            systemic_support=random.uniform(0.5, 1.0)
        )
        self.members.append(new_member)

    def cycle(self, intervention_effort):
        # Sort by openness
        candidates = sorted(
            [m for m in self.members if not m.rehabilitated],
            key=lambda m: m.openness, reverse=True
        )

                self.recruit()
